def drop_high_missing_columns(dfs, threshold=50):
    for table, df in dfs.items():
        missing_percentage = df.isnull().mean() * 100
        cols_to_drop = missing_percentage[missing_percentage > threshold].index.tolist()

        if cols_to_drop:
            print(f"\n {table}: Dropping {len(cols_to_drop)} columns with more than {threshold}% missing values:")

            df.drop(columns=cols_to_drop, inplace=True)
        else:
            print(f" {table}: No columns dropped (All columns have less than {threshold}% missing values).")

def impute_and_visualize(dfs):
    fig, axes = plt.subplots(nrows=2, ncols=len(dfs), figsize=(5 * len(dfs), 10))

    for i, (table, df) in enumerate(dfs.items()):
        # Count missing values
        missing_counts_before = df.isnull().sum()

        # Plot missing values before imputation (only if there are missing values)
        if missing_counts_before.sum() > 0:
            sns.barplot(x=missing_counts_before.index, y=missing_counts_before.values, ax=axes[0, i], color='crimson')
            axes[0, i].set_xticklabels(df.columns, rotation=90)
            axes[0, i].set_title(f"Before Imputation - {table}")
            axes[0, i].set_ylabel("Missing Count")
        else:
            axes[0, i].set_visible(False)

        # Impute missing values
        for col in df.columns:
            if df[col].dtype in ["int64", "float64"]:  # Numerical columns
                df[col].fillna(df[col].mean(), inplace=True)  # Mean Imputation
            else:  # Categorical columns
                mode_value = df[col].mode().dropna()
                if not mode_value.empty:
                    df[col].fillna(mode_value[0], inplace=True)  # Mode Imputation

        print(f" {table}: Missing values imputed using mean/mode.")
        print(df.head(), "\n")  # Display first 5 rows of cleaned table

        # Count missing values after imputation
        missing_counts_after = df.isnull().sum()

        # Plot missing values after imputation (only if there were missing values before)
        if missing_counts_before.sum() > 0:
            sns.barplot(x=missing_counts_after.index, y=missing_counts_after.values, ax=axes[1, i], color='green')
            axes[1, i].set_xticklabels(df.columns, rotation=90)
            axes[1, i].set_title(f"After Imputation - {table}")
            axes[1, i].set_ylabel("Missing Count")
        else:
            axes[1, i].set_visible(False)

    plt.tight_layout()
    plt.show()

# Apply function
impute_and_visualize(dfs)

def one_hot_encode(dfs):
    encoded_dfs = {}  # Dictionary to store transformed DataFrames

    for table, df in dfs.items():
        # Identify categorical columns
        categorical_cols = df.select_dtypes(include=['object']).columns.tolist()

        # Apply One-Hot Encoding
        df_encoded = pd.get_dummies(df, columns=categorical_cols, drop_first=True)

        # Store the transformed DataFrame
        encoded_dfs[table] = df_encoded

        print(f" {table}: Categorical columns encoded using One-Hot Encoding.")

def detect_outliers_zscore(df, threshold=3):
    z_scores = np.abs(zscore(df.select_dtypes(include=[np.number])))  # Only consider numeric columns
    outliers = (z_scores > threshold)  # True for outliers, False for non-outliers
    return outliers

def remove_outliers(df, outliers):
    # Remove rows that contain outliers
    df_no_outliers = df[~outliers.any(axis=1)]
    return df_no_outliers

def cap_outliers(df, outliers):
    # Cap the outliers by setting them to the threshold value (e.g., 5th and 95th percentiles)
    df_capped = df.copy()
    for col in df.select_dtypes(include=[np.number]).columns:
        lower_limit = df[col].quantile(0.05)
        upper_limit = df[col].quantile(0.95)
        df_capped[col] = np.where(outliers[col], df[col].clip(lower=lower_limit, upper=upper_limit), df[col])
    return df_capped

def impute_outliers(df, outliers, method='median'):
    # Impute the outliers with the median or mean of the column
    df_imputed = df.copy()
    for col in df.select_dtypes(include=[np.number]).columns:
        if method == 'median':
            central_value = df[col].median()
        elif method == 'mean':
            central_value = df[col].mean()
        df_imputed[col] = np.where(outliers[col], central_value, df[col])
    return df_imputed


def transform_outliers(df):
    # Apply log transformation to reduce the impact of extreme values
    df_transformed = df.copy()
    for col in df.select_dtypes(include=[np.number]).columns:
        df_transformed[col] = np.log1p(df_transformed[col])  # log1p handles zeroes in the data
    return df_transformed

    for table, df in dfs.items():

        if df.empty:

            cleaned_dfs[table] = df
            continue

        # Drop fully empty columns (optional)
        df.dropna(axis=1, how='all', inplace=True)

        # Convert numeric columns stored as strings
        for col in df.columns:
            # Remove unwanted characters and convert to numeric
            df[col] = df[col].astype(str).str.replace(',', '').str.replace('%', '').str.strip()

        # Store cleaned DataFrame
        cleaned_dfs[table] = df

        # Print sample summary
        print(df.dtypes)
        print(df.describe())  # Display summary statistics

    return cleaned_dfs

Examples
--------

    The simplest way to use nbconvert is

            > jupyter nbconvert mynotebook.ipynb --to html

            Options include ['asciidoc', 'custom', 'html', 'latex', 'markdown', 'notebook', 'pdf', 'python', 'qtpdf', 'qtpng', 'rst', 'script', 'slides', 'webpdf'].

            > jupyter nbconvert --to latex mynotebook.ipynb

            Both HTML and LaTeX support multiple output templates. LaTeX includes
            'base', 'article' and 'report'.  HTML includes 'basic', 'lab' and
            'classic'. You can specify the flavor of the format used.

            > jupyter nbconvert --to html --template lab mynotebook.ipynb

            You can also pipe the output to stdout, rather than a file

            > jupyter nbconvert mynotebook.ipynb --stdout

            PDF is generated via latex

            > jupyter nbconvert mynotebook.ipynb --to pdf

            You can get (and serve) a Reveal.js-powered slideshow

            > jupyter nbconvert myslides.ipynb --to slides --post serve

            Multiple notebooks can be given at the command line in a couple of
            different ways:

            > jupyter nbconvert notebook*.ipynb
            > jupyter nbconvert notebook1.ipynb notebook2.ipynb

            or you can specify the notebooks list in a config file, containing::

                c.NbConvertApp.notebooks = ["my_notebook.ipynb"]
